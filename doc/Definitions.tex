% Definitions.tex - definitions

% 2017-11-10 AMR TODO: should these be capitalized already?
\newglossaryentry{Success}{
	name=Success,
	plural=Successes,
	description={
		When a Check succeeds
	}
}
\newglossaryentry{Failure}{
	name=Failure,
	description={
		When a Check fails
	}
}
\newglossaryentry{Critical}{
	name=Critical,
	description={
		Extreme Success or Failure on a check, resulting in an exagerated result
	}
}
\newglossaryentry{Luck}{
	name=Luck,
	description={
		Percent Chance of automatic and/or Critical Success/Failure
	}
}
\newglossaryentry{Stat}{
	name=Stat,
	description={
		Basic building block of a Character
	}
}
\newglossaryentry{Modifier}{
	name=Modifier,
	description={
		A Change to a Value, usually Stat, Skill, or Check, be
		it temporary or otherwise; Does not effect Cost
	}
}
\newglossaryentry{Cost}{
	name=Cost,
	description={
		A value associated with a given Stat, Skill, Resource,
		etc. can be used to calculate a numerical `level' of a Character
	}
}
\newglossaryentry{Skill}{
	name=Skill,
	see={Stat},
	description={
		An attribute built up of zero or more depedant Stats or
		Skills which averages out to calculate a `default'
		value for this skill, and an associated Cost to raise this
		particular value
	}
}
\newglossaryentry{Check}{
	name=Check,
	see={Modifier,Skill,Stat},
	description={
		A collection of Modifiers and Skills or Stats,
		sometimes opposed, combined together to create a Chance of Success
		or Failure and then rolled against
	}
}
\newglossaryentry{Resource}{
	name=Resource,
	description={
		A Value (and associated Cost) usually with a Stat or Skill base
	}
}
\newglossaryentry{gls-GM}{
	name={Game Master},
	see={GM,Campaign},
	description={
		The story teller and/or god of a Campaign.  They make all the
		final calls of Success or Failure and will determine
		how the results of Checks actually effect the game world
	}
}
\newglossaryentry{Campaign}{
	name=Campaign,
	see={Character,gls-GM},
	description={
		A series of Events that happen to a collection of Characters
		usually overseen by one or more GMs
	}
}
\newglossaryentry{Trait}{
	name=Trait,
	see={Event,Modifier},
	description={
		An attribute to a character that is non-numerical in nature, will
		usually trigger and/or be triggered by one or more Events
		and generally affect Modifiers
	}
}
\newglossaryentry{Template}{
	name=Template,
	description={
		A collection of Stats, Skills, and Traits
		in an easier to handle single piece
	}
}
\newglossaryentry{Character}{
	name=Character,
	see={gls-GM},
	description={
		An entity in a game world, some are Player controlled and some are
		GM controlled
	}
}
\newglossaryentry{Event}{
	name=Event,
	description={
		Something that happens during normal progress of the story in a game
		world, and will usually effect Character in some way
	}
}

\makeglossaries

\newacronym{RPG}{RPG}{Role Playing Game}
\newacronym[see={[Glossary:]{gls-GM}}]{GM}{GM}{Game Master}
